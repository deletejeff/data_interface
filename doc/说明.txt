centOS不能使用ifconfig、netstat命令
安装包
yum install net-tools

编辑/etc/profile
vi /etc/profile
export JAVA_HOME=/home/mc/jdk1.8.0_211
export PATH=$PATH:$JAVA_HOME/bin


解决方案：
mysql8修改低版本mysql_native_password加密
    ALTER USER 'root'@'localhost' IDENTIFIED BY 'password' PASSWORD EXPIRE NEVER; #修改加密规则 
   ALTER USER 'root'@'localhost' IDENTIFIED WITH mysql_native_password BY 'password'; #更新一下用户的密码 

   FLUSH PRIVILEGES; #刷新权限 

再重置下密码：alter user 'root'@'localhost' identified by 'root';


create user 'root'@'%' identified by 'root1234';

create user 'xichu'@'%' identified by 'xichu123';

create database xichu;

//grant privileges on xichu.* to 'xichu'@'%' IDENTIFIED BY 'xichu123';
grant all privileges on xichu.* to 'xichu'@'%';
flush privileges;
